\begin{eqnarray*}
 	x_1 &=& r\cos(\phi_1) \\ 
	x_2 &=& r\sin(\phi_1)\cos(\phi_2) \\
	x_3 &=& r\sin(\phi_1)\sin(\phi_2)\cos(\phi_3) \\
	\cdots\\
	x_{n-1} &=& r\sin(\phi_1)\cdots\sin(\phi_{n-2})\cos(\phi_{n-1}) \\
	x_n &=& r\sin(\phi_1)\cdots\sin(\phi_{n-2})\sin(\phi_{n-1}) \\
	\phi_{n-1} &=& 0..2*\pi\\
	\forall {i=1..{n-1}}\phi_i &=& 0..\pi\\
\end{eqnarray*}
